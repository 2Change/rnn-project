def build_model():
    # create the base pre-trained model
       
    rnn = Sequential()
    rnn.add(LSTM(1024, input_shape=(X_train.shape[1:]), recurrent_dropout=0.25))
    rnn.add(Dense(512, activation='relu'))
    rnn.add(Dropout(0.5))
    rnn.add(Dense(nb_classes, activation='softmax'))

    return rnn
