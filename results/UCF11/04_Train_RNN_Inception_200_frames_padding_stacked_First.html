from keras.applications.inception_v3 import InceptionV3
from keras.applications.densenet import DenseNet121
from keras.models import Model
from keras import backend as K

def build_model():
       
    rnn = Sequential()
    rnn.add(LSTM(1024, input_shape=(X_train.shape[1:]), recurrent_dropout=0.2))
    rnn.add(Dense(512, activation='relu'))
    rnn.add(Dropout(0.5))
    rnn.add(Dense(nb_classes, activation='softmax'))

    return rnn


def build_model_stacked():
       
    rnn = Sequential()
    rnn.add(LSTM(128, input_shape=(X_train.shape[1:]), recurrent_dropout=0.2, return_sequences=True))
    rnn.add(LSTM(128, input_shape=(X_train.shape[1:])))
    rnn.add(Dense(512, activation='relu'))
    rnn.add(Dense(nb_classes, activation='softmax'))

    return rnn
